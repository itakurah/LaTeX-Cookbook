\setRecipeMeta{Soboro Don}{4}{30 MIN}{20 MIN}{./images/soboro.jpg}

\begin{recipe}
  \begin{ingredients}
    \ingredient{200g cooked rice}
    \ingredient{250g thinly sliced beef strips or 250g ground beef}
    \ingredient{1/2 spring onion}
    \ingredient{1 clove garlic}
    \ingredient{2 tbsp mirin}
    \ingredient{1 tbsp rice vinegar}
    \ingredient{1/2 tbsp sesame oil}
    \ingredient{3 tbsp light soy sauce}
    \ingredient{2 tbsp water}
    \ingredient{2 tbsp sugar}
    \ingredient{1 egg yolk}
  \end{ingredients}
  \begin{steps}
    \step{Prepare Rice: Cook rice in a rice cooker or following your usual method.}
    \step{Make Sesame Sauce: Mix 0.5 tbsp sesame oil, 1 tbsp rice vinegar, 2 tbsp soy sauce, 1 tbsp mirin, 1 tbsp sugar, 2 tbsp water, and a pinch of salt. Set aside.}
    \step{Cook Beef: Saut√© minced garlic for 2 min; add beef with 1 tbsp each of mirin, soy sauce, and sugar. Then cook until done (5--6 min).}
    \step{Assemble: Serve rice in a bowl, top with beef, egg yolk, and sliced green onions. Pour sauce over everything and mix well.}
  \end{steps}
\end{recipe}
