\begin{recipe}
  \begin{ingredients}
    \ingredient{5–6 medium onions}
    \ingredient{3-4 tbsp Sunflower oil}
    \ingredient{1.5 kg ground beef}
    \ingredient{1/2 tsp turmeric}
    \ingredient{3 heaped tsp ginger powder}
    \ingredient{1 tsp salt}
    \ingredient{3 tsp tomato paste}
    \ingredient{250g strained tomatoes (passata)}
    \ingredient{3/4 packet Shan Kofta seasoning mix}
    \ingredient{2 tsp dried oregano}
    \ingredient{4 small chili peppers, chopped}
    \ingredient{1 whole head of garlic, minced}
    \ingredient{1/2 packet Knorr Napoli sauce}
    \ingredient{1 cup Water}
  \end{ingredients}

  \begin{steps}
    \step{Prepare the onions and garlic: Finely chop the onions and mince the garlic; set aside.}
    \step{Sauté the onions: Heat sunflower oil in a large skillet over medium heat. Add the chopped onions and cook until they are rich brown and caramelized.}
    \step{Add meat and spices: Add the ground beef to the skillet. Stir in Shan Kofta seasoning, turmeric, and ginger powder. Mix thoroughly.}
    \step{Add chili and garlic: After 5 minutes, add chopped chili peppers and minced garlic. Cook until the garlic becomes fragrant.}
    \step{Season and reduce liquids: Add salt and continue cooking until most of the liquid from the beef evaporates.}
    \step{Add tomatoes and simmer: Increase the heat slightly and stir in the tomato paste, strained tomatoes, and about 1 cup of water. Bring to a boil and mix well.}
    \step{Simmer: Turn off the heat, cover the skillet, and let the mixture simmer for 10 minutes to allow flavors to blend.}
    \step{Serve with sauce: In a separate pot, heat the Knorr Napoli sauce with a bit of water. Add about 200g of the cooked Keema mixture, mix, and warm through.}
  \end{steps}
\end{recipe}
