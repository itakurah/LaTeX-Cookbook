\setRecipeMeta{Ragù}{4}{30 MIN}{20 MIN}{./images/ragu.jpg}

\begin{recipe}
  \begin{ingredients}
    \ingredient{500g pasta (Paccheri)}
    \ingredient{500g ground beef}
    \ingredient{400g passata di pomodoro (tomato puree)}
    \ingredient{100g water}
    \ingredient{Parmesan}
    \ingredient{1/2 glass dry red wine}
    \ingredient{3 carrots}
    \ingredient{2 onions}
    \ingredient{1 celery stalk}
    \ingredient{Olive oil}
  \end{ingredients}
  \begin{steps}
    \step{Prepare Vegetables: Chop carrots, onions, and celery stalk, then blend into small pieces.}
    \step{Sauté Vegetables: In a large pot, heat a bit of olive oil. Add the vegetable mix and cook until softened and slightly browned.}
    \step{Cook Beef: Add ground beef to the pot and cook until all liquid evaporates and a fond forms on the bottom.}
    \step{Deglaze with Wine: Pour in the red wine and cook until it has mostly evaporated.}
    \step{Add Tomatoes and Water: Stir in the passata di pomodoro (tomato puree) and water. Season with a bit of salt.}
    \step{Simmer: Cover partially and cook on medium-low heat for 2-4 hours, stirring occasionally. Adjust salt to taste before serving.}
  \end{steps}
\end{recipe}
