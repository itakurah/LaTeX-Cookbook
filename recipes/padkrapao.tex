\setRecipeMeta{Pad Kra Pao}{4}{15 MIN}{20 MIN}{./images/padkrapao.jpg}

\begin{recipe}
  \begin{ingredients}
    \ingredient{200g rice or 250g Asia noodles}
    \ingredient{2 eggs}
    \ingredient{3 tablespoons vegetable oil}
    \ingredient{2 shallots}
    \ingredient{7 cloves garlic}
    \ingredient{3 Thai bird or Holland chilies}
    \ingredient{450g ground beef or thinly sliced beef strips}
    \ingredient{1 tsp sugar}
    \ingredient{1 tbsp fish sauce}
    \ingredient{1 tbsp light soy sauce}
    \ingredient{1 tsp dark soy sauce}
    \ingredient{1 tsp oyster sauce}
    \ingredient{100ml chicken broth}
    \ingredient{Holy basil or Thai basil leaves}
  \end{ingredients}
  \begin{steps}
    \step{Start with Aromatics: Heat oil in a wok over medium-high heat. Add shallots and garlic, cooking for 3 minutes until crispy golden.}
    \step{Add Chilies: Toss in the chilies and cook for another minute.}
    \step{Cook the Meat: Increase the heat to high. Add the strips/ground beef, breaking it into small bits, and cook until browned and crisp.}
    \step{Add Sauces: Stir in sugar, fish sauce, light soy sauce, dark soy sauce, and oyster sauce, stir-fry for one minute.}
    \step{Deglaze: Add chicken broth, allowing it to evaporate.}
    \step{Finish with Basil: Stir in basil until wilted and serve over rice or Asia noodles. Optionally, top with a crispy fried egg.}
  \end{steps}
\end{recipe}
