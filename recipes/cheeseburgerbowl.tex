\setRecipeMeta{Cheeseburger Bowl}{2}{15 MIN}{30 MIN}{./images/cheeseburgerbowl.jpg}

\begin{recipe}
  \begin{ingredients}
    \ingredient{1 iceberg lettuce}
    \ingredient{2 red onions}
    \ingredient{200g cucumber}
    \ingredient{200g tomatoes}
    \ingredient{400g minced beef}
    \ingredient{300g potatoes}
    \ingredient{3 slices processed cheese}
    \ingredient{200ml plain yoghurt}
    \ingredient{2 tbsp mustard}
    \ingredient{2 tbsp light ketchup}
    \ingredient{2--4 tbsp pickle juice}
    \ingredient{6 small pickles}
  \end{ingredients}
  \begin{steps}
    \step{Peel and dice the potatoes. Roast them in the oven or air fryer at 200Â°C for about 30 minutes, until golden and crispy.}
    \step{Chop the lettuce, onions, cucumber, and tomatoes into bite-sized pieces.}
    \step{Fry the minced beef in a pan over medium-high heat for about 10 minutes, until browned and crispy. Season with salt and pepper.}
    \step{Add the processed cheese slices to the warm meat and stir until fully melted and combined.}
    \step{Prepare the sauce: finely chop the pickles and mix with the yoghurt, mustard, ketchup, and pickle juice. Adjust seasoning to taste.}
    \step{Combine all ingredients in a large bowl: layer the vegetables, top with the cheesy minced beef, and drizzle the sauce over everything. Serve warm.}
  \end{steps}
\end{recipe}
